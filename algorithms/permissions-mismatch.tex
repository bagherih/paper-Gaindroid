\begin{algorithmic}[1]
\Procedure{DetectPermMismatch}{app, graph, permMap}
\newline\Comment{\rlap{\bf Input:}\phantom{\textbf{Output:}} Decompiled APK, call/data flow graph, perm. map}
\newline\Comment{\textbf{Output:} List of detected mismatches}
    \State{dngrPerms $\gets$ \Call{GetDngrPermsFromManifest}{app}}
    \If{dngrPerms $=\emptyset$}
        \State{\Return{$\emptyset$}}
    \EndIf
    \State{callGraph $\gets$ \Call{BuildCallGraph}{app}}
    \If{app.targetSdkVersion $\geq 23$}
        \ForEachIn{method}{callGraph}
            \If {\Call{OverOnRequestPermsResult}{method}}
                \State{\Return{$\emptyset$}}    
            \EndIf
        \EndFor
    \EndIf
    \State{mm $\gets\emptyset$}
    \ForEachIn{method}{callGraph}
        \State {dfg $\gets$ \Call{GetDataFlowGraph}{graph, method}}
        \ForEachIn{block}{dfg}
            \ForEachIn{perm}{dngrPerms}
               	\If {permMap.\Call{IsUsingPerm}{perm, block}}
                   	\State{mm $\gets$ mm $\cup$ \{perm\}}
                \EndIf
            \EndFor
        \EndFor
    \EndFor
    \State{\Return{mm}}
\EndProcedure
\end{algorithmic}
